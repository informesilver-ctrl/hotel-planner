// ═══════════════════════════════════════════════════════════════════ // HOTEL PMS SYNC v2 — Google Apps Script // ═══════════════════════════════════════════════════════════════════ const CONFIG = { REMITENTE: 'noreply@ulysescloud.com', LABEL_OK: 'pms-procesado', CARPETA_DRIVE: 'Hotel PMS Data', ARCHIVO_JSON: 'pms_data.json', ARCHIVO_LOG: 'pms_sync_log.json', }; function procesarEmailsPMS() { const label = obtenerOCrearLabel(CONFIG.LABEL_OK); const carpeta = obtenerOCrearCarpeta(CONFIG.CARPETA_DRIVE); const query = `from:${CONFIG.REMITENTE} -label:${CONFIG.LABEL_OK}`; const threads = GmailApp.search(query); if (threads.length === 0) { console.log('No hay emails nuevos.'); return; } console.log(`${threads.length} emails nuevos de Septeo.`); let procesados = 0; for (const thread of threads) { for (const message of thread.getMessages()) { for (const att of message.getAttachments()) { const nombre = att.getName().toLowerCase(); if (!nombre.endsWith('.csv') && !nombre.endsWith('.xlsx')) continue; try { const texto = att.getDataAsString('UTF-8'); const datos = parsearCSVSepteo(texto); if (datos.length > 0) { guardarEnDrive(carpeta, datos, message.getDate()); procesados++; console.log(`✓ ${nombre} — ${datos.length} días`); } } catch(e) { console.error(`✗ Error: ${e.message}`); } } thread.addLabel(label); } } console.log(`Procesados: ${procesados}`); } // ═══════════════════════════════════════════════════════════════════ // PARSEAR CSV DE SEPTEO — formato real del informe Service Forecast // Columnas: Fecha, Reservada, % Occ, Pax, Llegada, Salida, CAME, // CUNA, DESAY, DESGR, DESMI, DESNI // ═══════════════════════════════════════════════════════════════════ function parsearCSVSepteo(texto) { const lineas = texto.split('\n').map(l => l.trim()).filter(l => l.length > 0); if (lineas.length < 2) return []; const sep = lineas[0].includes(';') ? ';' : ','; const headers = lineas[0].split(sep).map(h => h.trim().toLowerCase().replace(/['"]/g,'')); console.log('Headers:', headers.join(' | ')); const datos = []; for (let i = 1; i < lineas.length; i++) { const cols = lineas[i].split(sep).map(c => c.trim().replace(/['"]/g,'')); if (cols.length < 3) continue; const row = {}; headers.forEach((h, j) => { row[h] = cols[j] || '0'; }); // Detectar fecha — primera columna que tenga formato YYYY-MM-DD let fecha = null; for (const val of cols) { if (/^\d{4}-\d{2}-\d{2}$/.test(val)) { fecha = val; break; } } if (!fecha) continue; // Mapeo flexible — formato Septeo Service Forecast // Columnas conocidas: fecha, reservada, % occ, pax, llegada, salida, came, cuna, desay... const idx = { fecha: headers.indexOf('fecha'), reservada:headers.indexOf('reservada'), pct: headers.indexOf('% occ'), pax: headers.indexOf('pax'), llegada: headers.indexOf('llegada'), salida: headers.indexOf('salida'), desay: headers.indexOf('desay'), }; // Si no hay headers con nombre, usar posición fija del formato Septeo const usarPosicion = idx.reservada === -1; const item = { d: fecha, ocu: parseInt(usarPosicion ? cols[1] : row['reservada'] || row['hab_ocupadas'] || 0) || 0, pct: parseFloat(usarPosicion ? cols[2] : row['% occ'] || row['ocupacion_pct'] || 0) || 0, sal: parseInt(usarPosicion ? cols[5] : row['salida'] || row['salidas'] || 0) || 0, lle: parseInt(usarPosicion ? cols[4] : row['llegada'] || row['llegadas'] || 0) || 0, des: parseInt(usarPosicion ? cols[8] : row['desay'] || row['desayunos'] || 0) || 0, }; if (item.d) datos.push(item); } datos.sort((a, b) => a.d.localeCompare(b.d)); console.log(`Parseados ${datos.length} días. Primero: ${JSON.stringify(datos[0])}`); return datos; } // ═══════════════════════════════════════════════════════════════════ // GUARDAR EN DRIVE — acumula todos los meses en un solo JSON // ═══════════════════════════════════════════════════════════════════ function guardarEnDrive(carpeta, datos, fechaEmail) { if (datos.length === 0) return; const primerDia = datos[0].d; const mes = primerDia.slice(0, 7); // "2026-04" let todosMeses = {}; const iter = carpeta.getFilesByName(CONFIG.ARCHIVO_JSON); let archivo = null; if (iter.hasNext()) { archivo = iter.next(); try { todosMeses = JSON.parse(archivo.getBlob().getDataAsString()); } catch(e) {} todosMeses[mes] = datos; archivo.setContent(JSON.stringify(todosMeses)); } else { todosMeses[mes] = datos; archivo = carpeta.createFile(CONFIG.ARCHIVO_JSON, JSON.stringify(todosMeses), 'application/json'); archivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); } // Log actualizarLog(carpeta, { timestamp: new Date().toISOString(), mes, dias: datos.length, rango: `${primerDia} → ${datos[datos.length-1].d}`, muestra: datos[0], }); console.log(`✓ Drive actualizado: ${mes} con ${datos.length} días`); console.log(`URL: https://drive.google.com/uc?export=download&id=${archivo.getId()}`); } function actualizarLog(carpeta, entry) { let log = []; const iter = carpeta.getFilesByName(CONFIG.ARCHIVO_LOG); if (iter.hasNext()) { const f = iter.next(); try { log = JSON.parse(f.getBlob().getDataAsString()); } catch(e) {} log.unshift(entry); if (log.length > 100) log = log.slice(0, 100); f.setContent(JSON.stringify(log, null, 2)); } else { carpeta.createFile(CONFIG.ARCHIVO_LOG, JSON.stringify([entry], null, 2), 'application/json'); } } function obtenerOCrearLabel(nombre) { for (const l of GmailApp.getUserLabels()) if (l.getName()===nombre) return l; return GmailApp.createLabel(nombre); } function obtenerOCrearCarpeta(nombre) { const iter = DriveApp.getFoldersByName(nombre); if (iter.hasNext()) return iter.next(); return DriveApp.createFolder(nombre); } // ═══════════════════════════════════════════════════════════════════ // REPROCESAR — quita el label y vuelve a procesar todos los emails // ═══════════════════════════════════════════════════════════════════ function reprocesarTodos() { const label = obtenerOCrearLabel(CONFIG.LABEL_OK); const carpeta = obtenerOCrearCarpeta(CONFIG.CARPETA_DRIVE); const threads = GmailApp.search(`from:${CONFIG.REMITENTE} label:${CONFIG.LABEL_OK}`); console.log(`Reprocesando ${threads.length} emails...`); for (const thread of threads) { thread.removeLabel(label); } procesarEmailsPMS(); } function obtenerURLPublica() { const carpeta = obtenerOCrearCarpeta(CONFIG.CARPETA_DRIVE); const iter = carpeta.getFilesByName(CONFIG.ARCHIVO_JSON); if (!iter.hasNext()) { console.log('Archivo no existe aún.'); return; } const archivo = iter.next(); archivo.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); const url = `https://drive.google.com/uc?export=download&id=${archivo.getId()}`; console.log('URL para la app:', url); return url; } function probarConexion() { console.log('=== TEST ==='); const threads = GmailApp.search(`from:${CONFIG.REMITENTE}`); console.log(`Emails de Septeo: ${threads.length}`); const carpeta = obtenerOCrearCarpeta(CONFIG.CARPETA_DRIVE); console.log(`Carpeta Drive: ${carpeta.getName()}`); const iter = carpeta.getFilesByName(CONFIG.ARCHIVO_JSON); if (iter.hasNext()) { const f = iter.next(); const data = JSON.parse(f.getBlob().getDataAsString()); const meses = Object.keys(data); console.log(`Meses en Drive: ${meses.join(', ')}`); meses.forEach(m => console.log(` ${m}: ${data[m].length} días, muestra: ${JSON.stringify(data[m][0])}`)); console.log(`URL: https://drive.google.com/uc?export=download&id=${f.getId()}`); } else { console.log('Sin datos en Drive aún.'); } console.log('=== FIN TEST ==='); } // ═══════════════════════════════════════════════════════════════════ // ENDPOINT WEB — permite que la app lea los datos via fetch // Desplegar como "Aplicación web" en Apps Script // ═══════════════════════════════════════════════════════════════════ function doGet(e) { const carpeta = obtenerOCrearCarpeta(CONFIG.CARPETA_DRIVE); const iter = carpeta.getFilesByName(CONFIG.ARCHIVO_JSON); let datos = {}; if (iter.hasNext()) { try { datos = JSON.parse(iter.next().getBlob().getDataAsString()); } catch(e) {} } return ContentService .createTextOutput(JSON.stringify(datos)) .setMimeType(ContentService.MimeType.JSON); }
